function diagnose() { const u = document.getElementById('u-name').value.trim(); const p = document.getElementById('p-name').value.trim(); const btn = document.getElementById('fortune-btn'); // 未入力チェック if (!u || !p) { alert("お名前を両方入力してください。"); return; } // ひらがなのみ許可 const hiraRegex = /^[ぁ-ゖー　\s]+$/; if (!hiraRegex.test(u) || !hiraRegex.test(p)) { alert("✦ 星詠みの術式は、ひらがなの波長にのみ反応します。お名前はひらがなで綴ってください。 ✦"); return; } // 同名ブロック if (u === p) { alert("✦ ご自身と同じ影を結ぶことはできません。星が映し出すのは、異なる二つの魂の共鳴です。 ✦"); return; } // ボタンを「診断中...」に変更 const originalBtnText = btn.innerHTML; btn.disabled = true; btn.innerHTML = ` 星に問いかけ中... `; fetch(GAS_URL, { method: "POST", headers: { "Content-Type": "text/plain" }, body: JSON.stringify({ p1: u, p2: p }) }) .then(res => res.json()) .then(data => { btn.disabled = false; btn.innerHTML = originalBtnText; if (data.status === "success") { const res = data.result; let cls = "normal"; if ( res.includes("100") || res.includes("80") || res.includes("85") || res.includes("90") || res.includes("95") ) { cls = "good"; } else if ( res.includes("10") || res.includes("15") || res.includes("20") || res.includes("25") || res.includes("30") || res.includes("35") ) { cls = "bad"; } document.getElementById('setup-screen').classList.add('hidden'); const out = document.getElementById('result-output'); out.innerText = res; out.className = "result-main " + cls; document.getElementById('result-screen').classList.remove('hidden'); } else { alert("✦ お二人の『姓』と『名』、そのすべてが揃って初めて、星々は未来を紡ぎます。フルネームで正しく綴られていますか？ ✦"); } }) .catch(err => { btn.disabled = false; btn.innerHTML = originalBtnText; alert("通信が途絶えました。星の導きを再接続します。"); }); }
